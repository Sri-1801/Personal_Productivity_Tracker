#Data cleaning
import pandas as pd

# 1. Load the dataset
df = pd.read_excel("personal_productivity.xlsx")  # replace with your file name

# 2. View first few rows
print(df.head())

# 3. Check data types and missing values
print(df.info())
print(df.isnull().sum())

# 4. Convert 'Date' to datetime if needed
df['Date'] = pd.to_datetime(df['Date'], errors='coerce')

# 5. Check for outliers (basic example)
print(df.describe())

# 6. Handle any missing values (if found)
df = df.dropna()  # or df.fillna(value)

# 7. Save cleaned file
df.to_excel("personal_productivity_cleaned.xlsx", index=False)
print("Data cleaned and saved successfully!")

#EDA

import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns

# Load cleaned dataset
df = pd.read_excel("personal_productivity_cleaned.xlsx")

# 1. Hours Worked trend
plt.figure(figsize=(10,5))
sns.lineplot(x="Date", y="Hours_Worked", data=df, marker="o")
plt.title("Hours Worked Over Time")
plt.xticks(rotation=45)
plt.tight_layout()
plt.show()

# 2. Sleep Hours vs Mood Rating
plt.figure(figsize=(6,6))
sns.scatterplot(x="Sleep_Hours", y="Mood_Rating", data=df)
plt.title("Sleep Hours vs Mood Rating")
plt.tight_layout()
plt.show()

# 3. Learning Hours vs Tasks Completed
plt.figure(figsize=(6,6))
sns.scatterplot(x="Learning_Hours", y="Tasks_Completed", data=df)
plt.title("Learning Hours vs Tasks Completed")
plt.tight_layout()
plt.show()

# 4. Distribution of Mood Ratings
plt.figure(figsize=(8,5))
sns.histplot(df["Mood_Rating"], bins=10, kde=True)
plt.title("Distribution of Mood Ratings")
plt.tight_layout()
plt.show()

# 5. Top 5 Most Productive Days (by Tasks Completed)
top_days = df.sort_values(by="Tasks_Completed", ascending=False).head(5)
print("Top 5 Most Productive Days:")
print(top_days[["Date", "Tasks_Completed", "Hours_Worked"]])
